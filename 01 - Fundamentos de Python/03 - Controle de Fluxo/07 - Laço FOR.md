# Laço for


```python
# combinando for com a função range()


for numero in range(5):
    print(f'carregando #{numero}')

'''
carregando #0
carregando #1
carregando #2
carregando #3
carregando #4
'''


# combinado for com um iterável


clubes = ['Chelsea', 'United', 'City', 'Liverpool', 'Arsenal']

for clube in clubes:
    print(f'Clube: {clube}')

'''
Clube: Chelsea
Clube: United
Clube: City
Clube: Liverpool
Clube: Arsenal
'''
```
